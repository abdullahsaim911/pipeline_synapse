# Synapse Frontend — Requirements
# ─────────────────────────────────────────────────────────────────────────────
# Runtime : Node.js >= 18.x  (https://nodejs.org)
# Package  : npm (comes with Node)
#
# SETUP
#   1. Install Node 18+ if you haven't already.
#   2. In this folder, run:
#        npm install
#   3. Start the dev server:
#        npm run dev          → opens http://localhost:5173
#
# OPTIONAL — Electron desktop app
#   npm run electron:dev     → runs Vite + Electron together
#
# BUILD for production (web)
#   npm run build            → output goes to /dist
#
# ─────────────────────────────────────────────────────────────────────────────
# All dependencies below are managed by package.json / node_modules.
# You do NOT install these manually — npm install handles everything.
# ─────────────────────────────────────────────────────────────────────────────

# BACKEND CONFIGURATION  (src/api.js)
# ─────────────────────────────────────────────────────────────────────────────
# Line 1  — USE_MOCK = true
#   Change to false when connecting to the real backend.
#
# Line 3  — API_BASE_URL
#   Set this to the backend server URL (Cloudflare tunnel or local address).
#   Current value: "https://gorgeous-arms-day-relevance.trycloudflare.com"
#
# API endpoints the frontend calls:
#   POST /video/process          → starts processing a YouTube video
#   GET  /api/jobs/:jobId        → polls job progress
#   POST /api/tts                → fetches TTS audio + text explanation
#                                  body: { intervention_id, output_mode }
#   GET  /api/explanation/:id    → fetches text explanation (mode as query param)
